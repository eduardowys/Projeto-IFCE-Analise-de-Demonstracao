Dim_ContaDeposito = 
VAR ContasValidas =
    FILTER(
        DISTINCT( BASE_RECEBIMENTOS[CONTA] ),
        NOT( ISBLANK( BASE_RECEBIMENTOS[CONTA] ) )
    )
RETURN
ADDCOLUMNS(
    ContasValidas,
    "ID_Conta",
        RANKX(
            ContasValidas,
            BASE_RECEBIMENTOS[CONTA],
            ,
            ASC
        )
)
