Dim_FormaRecebimento =
VAR FormasValidas =
    FILTER(
        DISTINCT ( BASE_RECEBIMENTOS[FORMA_DE_RECEBIMENTO] ),
        NOT (
            ISBLANK ( BASE_RECEBIMENTOS[FORMA_DE_RECEBIMENTO] ) ||
            TRIM ( BASE_RECEBIMENTOS[FORMA_DE_RECEBIMENTO] ) = ""
        )
    )
RETURN
ADDCOLUMNS(
    FormasValidas,
    "ID_Forma",
        RANKX(
            FormasValidas,
            BASE_RECEBIMENTOS[FORMA_DE_RECEBIMENTO],
            ,
            ASC
        )
)
